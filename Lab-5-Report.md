``` r
iris_sum <-
  iris_cm %>%
  group_by(Species) %>%
  summarize(
    sample_size = n(),
    mean_w = mean(petal_width),
    str_dev = sd(petal_width),
    var = var(petal_width),
    sem = mean(petal_width) / sqrt(n()),

    ci_lower = mean_w - 2 * sem
```
